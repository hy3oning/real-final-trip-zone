DDL Summary

테이블군 요약

18+회원, 상품, 예약, 혜택, 운영, 문의 핵심 테이블
6주요 상태 모델: 회원, 판매자, 숙소, 객실, 예약, 결제/문의
OracleNUMBER(10), DATE, VARCHAR2 기반 스키마

Constraints

제약 / 인덱스 기준

  • 이메일 유니크, 권한 복합 유니크, 판매자 사업자번호 유니크
  • 날짜 범위는 CHECK_OUT_DATE > CHECK_IN_DATE 기준
  • 환불 금액은 결제 금액을 초과할 수 없다.
  • 회원 쿠폰 상태는 ISSUED / USED / EXPIRED / CANCELED만 허용한다.
  • 예약/결제/숙소/문의는 운영 조회용 인덱스를 별도로 둔다.

실행 기준 원문 일부

-- TripZone company-grade DDL v2
-- Oracle-style schema

CREATE TABLE MEMBER_GRADES (
    GRADE_NO NUMBER(10) NOT NULL,
    GRADE_CODE VARCHAR2(30) NOT NULL,
    GRADE_NAME VARCHAR2(50) NOT NULL,
    MIN_TOTAL_AMOUNT NUMBER(12) DEFAULT 0 NOT NULL,
    MIN_STAY_COUNT NUMBER(10) DEFAULT 0 NOT NULL,
    MILEAGE_RATE NUMBER(5,2) DEFAULT 0 NOT NULL,
    BENEFIT_DESC VARCHAR2(500),
    STATUS VARCHAR2(20) DEFAULT 'ACTIVE' NOT NULL,
    SORT_ORDER NUMBER(3) DEFAULT 1 NOT NULL,
    REG_DATE DATE DEFAULT SYSDATE NOT NULL,
    UPD_DATE DATE DEFAULT SYSDATE NOT NULL
);

CREATE TABLE USERS (
    USER_NO NUMBER(10) NOT NULL,
    USER_NAME VARCHAR2(100) NOT NULL,
    EMAIL VARCHAR2(100) NOT NULL,
    PHONE VARCHAR2(20) NOT NULL,
    GRADE_NO NUMBER(10),
    MILEAGE NUMBER(12) DEFAULT 0 NOT NULL,
    ENABLED CHAR(1) DEFAULT '1' NOT NULL,
    USER_STATUS VARCHAR2(20) DEFAULT 'ACTIVE' NOT NULL,
    DELETED_AT DATE
);

CREATE TABLE HOST_PROFILES (
    HOST_NO NUMBER(10) NOT NULL,
    USER_NO NUMBER(10) NOT NULL,
    BUSINESS_NAME VARCHAR2(100) NOT NULL,
    BUSINESS_NUMBER VARCHAR2(50) NOT NULL,
    OWNER_NAME VARCHAR2(100) NOT NULL,
    APPROVAL_STATUS VARCHAR2(20) DEFAULT 'PENDING' NOT NULL
);

CREATE TABLE LODGINGS (
    LODGING_NO NUMBER(10) NOT NULL,
    HOST_NO NUMBER(10) NOT NULL,
    LODGING_NAME VARCHAR2(200) NOT NULL,
    LODGING_TYPE VARCHAR2(50) NOT NULL,
    REGION VARCHAR2(100) NOT NULL,
    ADDRESS VARCHAR2(300) NOT NULL,
    LATITUDE NUMBER(10,7),
    LONGITUDE NUMBER(10,7),
    DESCRIPTION VARCHAR2(2000),
    CHECK_IN_TIME VARCHAR2(20),
    CHECK_OUT_TIME VARCHAR2(20),
    STATUS VARCHAR2(20) DEFAULT 'ACTIVE' NOT NULL
);

CREATE TABLE ROOMS (
    ROOM_NO NUMBER(10) NOT NULL,
    LODGING_NO NUMBER(10) NOT NULL,
    ROOM_NAME VARCHAR2(200) NOT NULL,
    ROOM_TYPE VARCHAR2(50) NOT NULL,
    MAX_GUEST_COUNT NUMBER(5) NOT NULL,
    PRICE_PER_NIGHT NUMBER(10) NOT NULL,
    STATUS VARCHAR2(20) DEFAULT 'AVAILABLE' NOT NULL
);

CREATE TABLE BOOKINGS (
    BOOKING_NO NUMBER(10) NOT NULL,
    USER_NO NUMBER(10) NOT NULL,
    ROOM_NO NUMBER(10) NOT NULL,
    CHECK_IN_DATE DATE NOT NULL,
    CHECK_OUT_DATE DATE NOT NULL,
    GUEST_COUNT NUMBER(5) NOT NULL,
    PRICE_PER_NIGHT NUMBER(10) NOT NULL,
    DISCOUNT_AMOUNT NUMBER(10) DEFAULT 0 NOT NULL,
    TOTAL_PRICE NUMBER(10) NOT NULL,
    STATUS VARCHAR2(20) DEFAULT 'PENDING' NOT NULL
);

CREATE TABLE PAYMENTS (
    PAYMENT_NO NUMBER(10) NOT NULL,
    BOOKING_NO NUMBER(10) NOT NULL,
    PAYMENT_METHOD VARCHAR2(30) NOT NULL,
    PG_PROVIDER VARCHAR2(50),
    PAYMENT_AMOUNT NUMBER(10) NOT NULL,
    REFUND_AMOUNT NUMBER(10) DEFAULT 0 NOT NULL,
    PAYMENT_STATUS VARCHAR2(30) DEFAULT 'READY' NOT NULL
);

CREATE TABLE INQUIRY_ROOMS (
    INQUIRY_ROOM_NO NUMBER(10) NOT NULL,
    USER_NO NUMBER(10) NOT NULL,
    HOST_NO NUMBER(10),
    LODGING_NO NUMBER(10),
    BOOKING_NO NUMBER(10),
    TITLE VARCHAR2(200) NOT NULL,
    INQUIRY_TYPE VARCHAR2(30) NOT NULL,
    ROOM_STATUS VARCHAR2(20) DEFAULT 'OPEN' NOT NULL
);

CREATE TABLE INQUIRY_MESSAGES (
    INQUIRY_MESSAGE_NO NUMBER(10) NOT NULL,
    INQUIRY_ROOM_NO NUMBER(10) NOT NULL,
    USER_NO NUMBER(10) NOT NULL,
    MESSAGE_CONTENT CLOB NOT NULL,
    IS_READ CHAR(1) DEFAULT '0' NOT NULL
);
